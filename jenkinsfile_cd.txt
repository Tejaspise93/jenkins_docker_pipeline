pipeline {
    agent any

    environment {
        DOCKERHUB_REPO = "<docker repo>/myweb"
        DEPLOYMENT     = "myweb"
        NAMESPACE      = "default"
        GIT_REPO       = "<git hub repo>"
    }

    parameters {
        string(name: 'IMAGE_TAG', defaultValue: '', description: 'Docker image tag to deploy (from pom.xml version)')
        string(name: 'BRANCH', defaultValue: 'master', description: 'Branch to pull K8s manifests from')
    }

    stages {

        stage('Validate Inputs') {
            steps {
                script {
                    if (!params.IMAGE_TAG?.trim()) {
                        error "IMAGE_TAG parameter is required. Pass the version from pom.xml (e.g. 0.0.9)"
                    }
                    echo "--- Deploying ${DOCKERHUB_REPO}:${params.IMAGE_TAG} to EKS ---"
                }
            }
        }

        stage('Checkout K8s Manifests') {
            steps {
                git branch: "${params.BRANCH}",
                    url: "${GIT_REPO}"
            }
        }

        stage('Deploy to EKS') {
            steps {
                withCredentials([file(credentialsId: 'k8s-kubeconfig', variable: 'KUBECONFIG')]) {
                    sh """
                        echo "--- Updating image tag in manifests ---"
                        sed -i 's|IMAGE_TAG_PLACEHOLDER|${params.IMAGE_TAG}|g' deployment.yaml
                        echo "--- Applying K8s manifests ---"
                        kubectl --kubeconfig=\$KUBECONFIG apply -f deployment.yaml -n ${NAMESPACE}
                        kubectl --kubeconfig=\$KUBECONFIG apply -f service.yaml -n ${NAMESPACE}
                        echo "--- Waiting for rollout to complete ---"
                        kubectl --kubeconfig=\$KUBECONFIG rollout status deployment/${DEPLOYMENT} \
                            -n ${NAMESPACE} \
                            --timeout=300s
                    """
                }
            }
        }

        stage('Verify Deployment') {
            steps {
                withCredentials([file(credentialsId: 'k8s-kubeconfig', variable: 'KUBECONFIG')]) {
                    sh """
                        echo "--- Pod Status ---"
                        kubectl --kubeconfig=\$KUBECONFIG get pods -l app=myweb -n ${NAMESPACE} -o wide

                        echo "--- Service Info ---"
                        kubectl --kubeconfig=\$KUBECONFIG get svc myweb-service -n ${NAMESPACE}

                        echo "--- Deployment Details ---"
                        kubectl --kubeconfig=\$KUBECONFIG describe deployment ${DEPLOYMENT} -n ${NAMESPACE}
                    """
                }
            }
        }

    }

    post {
        success {
            echo "--- Deployment successful ---"
            withCredentials([file(credentialsId: 'k8s-kubeconfig', variable: 'KUBECONFIG')]) {
                script {
                    def url = sh(
                        script: "kubectl --kubeconfig=\$KUBECONFIG get svc myweb-service -n ${NAMESPACE} -o jsonpath='{.status.loadBalancer.ingress[0].hostname}'",
                        returnStdout: true
                    ).trim()
                    echo "--- Service Endpoint ---"
                    echo "App is accessible at: http://${url}/myweb"
                }
            }
        }
        failure {
            echo "--- Deployment FAILED — initiating rollback ---"
            withCredentials([file(credentialsId: 'k8s-kubeconfig', variable: 'KUBECONFIG')]) {
                sh """
                    kubectl --kubeconfig=\$KUBECONFIG rollout undo deployment/${DEPLOYMENT} -n ${NAMESPACE} || echo "--- No previous deployment to rollback to ---"
                    echo "--- Rollback complete. Previous version restored ---"
                    kubectl --kubeconfig=\$KUBECONFIG rollout status deployment/${DEPLOYMENT} \
                        -n ${NAMESPACE} --timeout=300s || true
                """
            }
        }
        always {
            cleanWs()
        }
    }
}
